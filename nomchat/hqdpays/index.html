payment.js 
// Initialize hQd Pays client
import { HqdPays } from '@nomchat/hqdpays';

const client = new HqdPays({
  apiKey: process.env.HQDPAYS_KEY
});

// Create a payment
const payment = await client.create({
  amount:      299,
  currency:    'RUB',
  user_id:     'nomchat_user_id',
  description: 'Pro Credits Pack'
});

// Handle result
if (payment.status === 'success') {
  console.log(`Paid: ₽${payment.amount}`);
}
